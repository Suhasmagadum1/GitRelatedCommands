# GitRelatedCommands

    # Adding files in local
    git add .
    # Commit files in local
    git commit -m "<msg>"
    
    # push changes to remote
    git push <branch_name>
    
    # change to ssh to http url 
    git remote set-url origin https://...
    git remote set-url --push origin https://...
    
    # check url 
    git remote -v
